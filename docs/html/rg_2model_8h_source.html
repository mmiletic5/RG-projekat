119  void loadTextureMaterial(aiMaterial* mat, aiTextureType type, std::string typeName,
120  std::vector<Texture> & textures) {
121 
122  for (unsigned int i = 0; i < mat->GetTextureCount(type); ++i) {

125 
126  bool skip = false;
127 
128  for (unsigned int j = 0; j < loaded_textures.size(); ++j) {

149 unsigned int TextureFromFile(const char* filename, std::string directory) {
150  std::string fullPath(directory + "/" + filename);
151 
152  unsigned int textureID;
153  glGenTextures(1, &textureID);
154 
155  int width, height, nrComponents;
156  unsigned char* data = stbi_load(fullPath.c_str(), &width, &height, &nrComponents, 0);
157  if (data) {
158  GLenum format;
159  if (nrComponents == 1) {
160  format = GL_RED;
161  } else if (nrComponents == 3) {
162  format = GL_RGB;
163  } else if (nrComponents == 4) {
164  format = GL_RGBA;
165  }
166  glBindTexture(GL_TEXTURE_2D, textureID);
167  glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
168  glGenerateMipmap(GL_TEXTURE_2D);
169 
170  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
171  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
172  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
173  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAX_FILTER, GL_LINEAR);
174 
175  } else {
176  ASSERT(false, "Failed to load texture image");
177  }
178  stbi_image_free(data);
179  return textureID;
180 }
