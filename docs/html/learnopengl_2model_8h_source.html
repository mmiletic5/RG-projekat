75  // processes a node in a recursive fashion. Processes each individual mesh located at the node and repeats this process on its children nodes (if any).
76  void processNode(aiNode *node, const aiScene *scene)
77  {
78  // process each mesh located at the current node
79  for(unsigned int i = 0; i < node->mNumMeshes; i++)
80  {
81  // the node object only contains indices to index the actual objects in the scene.
82  // the scene contains all the data, node is just to keep stuff organized (like relations between nodes).
83  aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
84  meshes.push_back(processMesh(mesh, scene));
85  }
86  // after we've processed all of the meshes (if any) we then recursively process each of the children nodes
87  for(unsigned int i = 0; i < node->mNumChildren; i++)
88  {
89  processNode(node->mChildren[i], scene);

94  Mesh processMesh(aiMesh *mesh, const aiScene *scene)
95  {
96  // data to fill
97  vector<Vertex> vertices;
98  vector<unsigned int> indices;
99  vector<Texture> textures;
100 
101  // walk through each of the mesh's vertices
102  for(unsigned int i = 0; i < mesh->mNumVertices; i++)
103  {
104  Vertex vertex;
105  glm::vec3 vector; // we declare a placeholder vector since assimp_ uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
106  // positions
107  vector.x = mesh->mVertices[i].x;
108  vector.y = mesh->mVertices[i].y;
109  vector.z = mesh->mVertices[i].z;
110  vertex.Position = vector;
111  // normals
112  if (mesh->HasNormals())
113  {
114  vector.x = mesh->mNormals[i].x;
115  vector.y = mesh->mNormals[i].y;
116  vector.z = mesh->mNormals[i].z;
117  vertex.Normal = vector;
118  }
119  // texture coordinates
120  if(mesh->mTextureCoords[0]) // does the mesh contain texture coordinates?
121  {
122  glm::vec2 vec;
123  // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
124  // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
125  vec.x = mesh->mTextureCoords[0][i].x;
126  vec.y = mesh->mTextureCoords[0][i].y;
127  vertex.TexCoords = vec;
128  // tangent
129  vector.x = mesh->mTangents[i].x;
130  vector.y = mesh->mTangents[i].y;
131  vector.z = mesh->mTangents[i].z;
132  vertex.Tangent = vector;
133  // bitangent
134  vector.x = mesh->mBitangents[i].x;
135  vector.y = mesh->mBitangents[i].y;
136  vector.z = mesh->mBitangents[i].z;
137  vertex.Bitangent = vector;
138  }
139  else
140  vertex.TexCoords = glm::vec2(0.0f, 0.0f);
141 
142  vertices.push_back(vertex);
143 
144 
145  }
146  // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
147  for(unsigned int i = 0; i < mesh->mNumFaces; i++)
148  {
149  aiFace face = mesh->mFaces[i];
150  // retrieve all indices of the face and store them in the indices vector
151  for(unsigned int j = 0; j < face.mNumIndices; j++)
152  indices.push_back(face.mIndices[j]);
153  }
154  // process materials
155  aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
156  // we assume a convention for sampler names in the shaders. Each diffuse texture should be named
157  // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
158  // Same applies to other texture as the following list summarizes:
159  // diffuse: texture_diffuseN
160  // specular: texture_specularN
161  // normal: texture_normalN
162  aiColor3D color(0.0f, 0.0f, 0.0f);
163  material->Get(AI_MATKEY_COLOR_AMBIENT, color);
164 
165 
166  // 1. diffuse maps
167  vector<Texture> diffuseMaps = loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
168  textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
169  // 2. specular maps
170  vector<Texture> specularMaps = loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
171  textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
172  // 3. normal maps
173  std::vector<Texture> normalMaps = loadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
174  textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
175  // 4. height maps
176  std::vector<Texture> heightMaps = loadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
177  textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
178 
179 
180 
181  // return a mesh object created from the extracted mesh data
182  return Mesh(vertices, indices, textures);

185  // checks all material textures of a given type and loads the textures if they're not loaded yet.
186  // the required info is returned as a Texture struct.
187  vector<Texture> loadMaterialTextures(aiMaterial *mat, aiTextureType type, string typeName)
188  {
189  vector<Texture> textures;
190  for(unsigned int i = 0; i < mat->GetTextureCount(type); i++)
191  {
192  aiString str;
193  mat->GetTexture(type, i, &str);
194  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
195  bool skip = false;
196  for(unsigned int j = 0; j < textures_loaded.size(); j++)
197  {
198  if(std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
199  {
200  textures.push_back(textures_loaded[j]);
201  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
202  break;
203  }
204  }
205  if(!skip)
206  { // if texture hasn't been loaded already, load it
207  Texture texture;
208  texture.id = TextureFromFile(str.C_Str(), this->directory);
209  texture.type = typeName;
210  texture.path = str.C_Str();
211  textures.push_back(texture);
212  textures_loaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
213  }
214  }
215  return textures;
216  }
