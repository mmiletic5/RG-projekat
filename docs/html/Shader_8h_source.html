19  Shader(std::string vertexShaderPath, std::string fragmentShaderPath) {
20  appendShaderFolderIfNotPresent(vertexShaderPath);
21  appendShaderFolderIfNotPresent(fragmentShaderPath);
22  // build and compile our shader program
23  // ------------------------------------
24  // vertex shader
25  std::string vsString = readFileContents(vertexShaderPath);
26  ASSERT(!vsString.empty(), "Vertex shader source is empty!");
27  const char* vertexShaderSource = vsString.c_str();
28  int vertexShader = glCreateShader(GL_VERTEX_SHADER);
29  glShaderSource(vertexShader, 1, &vertexShaderSource, NULL);
30  glCompileShader(vertexShader);
31  // check for shader compile errors
32  int success;
33  char infoLog[512];
34  glGetShaderiv(vertexShader, GL_COMPILE_STATUS, &success);
35  if (!success)
36  {
37  glGetShaderInfoLog(vertexShader, 512, NULL, infoLog);
38  std::cout << "ERROR::SHADER::VERTEX::COMPILATION_FAILED\n" << infoLog << std::endl;
39  }
40  // fragment shader
41  int fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
42  std::string fsString = readFileContents(fragmentShaderPath);
43  ASSERT(!fsString.empty(), "Fragment shader empty!");
44  const char* fragmentShaderSource = fsString.c_str();
45  glShaderSource(fragmentShader, 1, &fragmentShaderSource, NULL);
46  glCompileShader(fragmentShader);
47  // check for shader compile errors
48  glGetShaderiv(fragmentShader, GL_COMPILE_STATUS, &success);
49  if (!success)
50  {
51  glGetShaderInfoLog(fragmentShader, 512, NULL, infoLog);
52  std::cout << "ERROR::SHADER::FRAGMENT::COMPILATION_FAILED\n" << infoLog << std::endl;
53  }
54  // link shaders
55  int shaderProgram = glCreateProgram();
56  glAttachShader(shaderProgram, vertexShader);
57  glAttachShader(shaderProgram, fragmentShader);
58  glLinkProgram(shaderProgram);
59  // check for linking errors
60  glGetProgramiv(shaderProgram, GL_LINK_STATUS, &success);
61  if (!success) {
62  glGetProgramInfoLog(shaderProgram, 512, NULL, infoLog);
63  std::cout << "ERROR::SHADER::PROGRAM::LINKING_FAILED\n" << infoLog << std::endl;
64  }
65  glDeleteShader(vertexShader);
66  glDeleteShader(fragmentShader);
67  m_Id = shaderProgram;
68  }
